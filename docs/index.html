React Test

To get started, import it and use it:

import $ from 'react-test';

const Counter = () => {
  const [count, setCounter] = useState(0);
  return <div onClick={e => setCounter(count + 1)}>{count}<div>;
};

describe('Counter.js', () => {
  it('starts on 0', () => {
    expect($(<Counter />).text()).toBe('0');
  });
});

.attr()

Read the attributes for the matched elements

.attr('name')

.children()

Get the direct children of all of the nodes with an optional filter

.children(filter)

.click()

Calls a click event on all of the matched nodes

.click()

.closest()

Find the first ancestor that matches the selector for each node

.closest(filter)

.find()

Get all of the descendants of the nodes with an optional filter

.find(filter)

.html()

Retrieve the html of the elements:

.html()

.text()

Retrieve the textContent of the elements:

.text()

.trigger()

Trigger an event. Right now it only accepts 'click'.

.trigger()
